-- =============================================
-- SUPERSTORE SALES ANALYSIS - SQL QUERIES
-- =============================================

-- 1. Create Table (Main Table Structure)
CREATE TABLE superstore_sales (
    row_id          INTEGER,
    order_id        VARCHAR(50),
    order_date      VARCHAR(50),
    ship_date       VARCHAR(50),
    ship_mode       VARCHAR(50),
    customer_id     VARCHAR(50),
    customer_name   VARCHAR(100),
    segment         VARCHAR(50),
    country         VARCHAR(50),
    city            VARCHAR(100),
    state           VARCHAR(100),
    postal_code     VARCHAR(20),
    region          VARCHAR(50),
    product_id      VARCHAR(50),
    category        VARCHAR(50),
    sub_category    VARCHAR(50),
    product_name    TEXT,
    sales           NUMERIC,
    quantity        INTEGER,
    discount        NUMERIC,
    profit          NUMERIC
);

-- 2. Import Data (after moving CSV to C:\temp)
COPY superstore_sales 
FROM 'C:\temp\Sample - Superstore.csv'
DELIMITER ',' 
CSV HEADER 
ENCODING 'WIN1252';

-- 3. Add Clean Date Columns
ALTER TABLE superstore_sales 
ADD COLUMN IF NOT EXISTS order_date_clean DATE,
ADD COLUMN IF NOT EXISTS ship_date_clean DATE;

UPDATE superstore_sales 
SET order_date_clean = TO_DATE(order_date, 'MM/DD/YYYY'),
    ship_date_clean = TO_DATE(ship_date, 'MM/DD/YYYY');

-- 4. Business Analysis Queries

-- Query 1: Profit by Category
SELECT 
    category,
    ROUND(SUM(sales), 2) AS total_sales,
    ROUND(SUM(profit), 2) AS total_profit,
    ROUND(AVG(profit / NULLIF(sales, 0) * 100), 2) AS profit_margin_pct,
    COUNT(*) AS num_orders
FROM superstore_sales
GROUP BY category
ORDER BY total_profit DESC;

-- Query 2: Performance by Region
SELECT 
    region,
    ROUND(SUM(sales), 2) AS total_sales,
    ROUND(SUM(profit), 2) AS total_profit,
    COUNT(*) AS num_orders
FROM superstore_sales
GROUP BY region
ORDER BY total_profit DESC;

-- Query 3: Monthly Sales Trend
SELECT 
    TO_CHAR(order_date_clean, 'YYYY-MM') AS month_year,
    ROUND(SUM(sales), 2) AS total_sales,
    ROUND(SUM(profit), 2) AS total_profit
FROM superstore_sales
GROUP BY month_year
ORDER BY month_year;

-- Quick Summary
SELECT 
    COUNT(*) as total_orders,
    ROUND(SUM(sales), 2) as total_sales,
    ROUND(SUM(profit), 2) as total_profit
FROM superstore_sales;
